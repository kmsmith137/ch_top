# Paths the sandboxed agent must NOT see (goal 3: deny, not even read).
#
# Each existing path listed here is masked by mounting an empty, unreadable
# (chmod 000) node over it -- an empty dir over a directory, an empty file over a
# file -- so inside the container the agent finds an empty placeholder instead of
# the contents. Read by sbox-claude at every launch --
# edit freely; changes take effect on the next launch, for all worktrees in the
# group, with no re-render. (The launcher reads this copy from the TOPLEVEL ch_dev,
# and it is mounted read-only into the container, so the agent cannot edit it.)
#
# Syntax: one path per line. A leading ~ (or $HOME) expands to your home dir.
# Globs are expanded. Lines starting with # and blank lines are ignored.
# Non-existent paths are silently skipped. Everything not listed here and not
# granted read-write by readwrite.txt is already read-only.

~/.ssh
~/.pypirc
~/.aws
~/.gnupg
~/.netrc
~/.git-credentials

# The whole ~/.config dir can hold secrets (gh tokens, etc.), so mask it all. If
# a build/test turns out to need a *non-secret* tool config that lives there
# (e.g. pip, matplotlib), delete this line and list the specific secret subpaths
# instead, leaving the harmless ones readable.
~/.config

# Your personal Claude state. The sandbox uses its own config dir
# (CLAUDE_CONFIG_DIR = the grouping dir, e.g. ~/ch/.claude.json + .credentials.json
# + projects/), so it does NOT need these -- masking them keeps your live OAuth
# token and host chat history out of the agent's reach. (Auth is therefore
# per-group: run '/login' once per grouping dir. See README.md "Auth".)
~/.claude
~/.claude.json
